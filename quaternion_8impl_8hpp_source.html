68 template<typename T>
69 Euler<T> Quaternion<T>::euler() const
70 {
71  Euler<T> euler;
72 
73  T sqx = x * x;
74  T sqy = y * y;
75  T sqz = z * z;
76  T sqw = w * w;
77 
78  T sarg = -2 * (x * z - w * y) / (sqx + sqy + sqz + sqw);
79  if (sarg <= -0.99999) {
80  euler.roll = make_radian(0.0);
81  euler.pitch = make_radian(-0.5_pi);
82  euler.yaw = -2.0 * signed_arctan(y, x);
83  } else if (sarg >= 0.99999) {
84  euler.roll = make_radian(0.0);
85  euler.pitch = make_radian(0.5_pi);
86  euler.yaw = 2.0 * signed_arctan(y, x);
87  } else {
88  euler.roll = signed_arctan(2 * (y * z + w * x), sqw - sqx - sqy + sqz);
89  euler.pitch = arcsin(sarg);
90  euler.yaw = signed_arctan(2 * (x * y + w * z), sqw + sqx - sqy - sqz);
91  }
92 
93  euler.roll = euler.roll.normalize();
94  euler.pitch = euler.pitch.normalize();
95  euler.yaw = euler.yaw.normalize();
96 
97  return euler;
98 }
